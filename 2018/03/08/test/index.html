from PIL import Image

im = Image.open ('1.bmp')
im2 = im.copy()

pix = im2.load()
width,heigth = im2.size

for x in xrange(0,width):
	for y in xrange(0,heigth):
		if pix[x,y]&0x1==0:
			pix[x,y]=0
		else:
			pix[x,y]=255
im2.save('2.png')
print 'ok'
